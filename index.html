async function askBot() { const inputField = document.getElementById("userInput"); const log = document.getElementById("chat-window"); const input = inputField.value.trim(); if (!input) return; log.innerHTML += `
${input}
`; inputField.value = ""; const loadingId = "load-" + Date.now(); log.innerHTML += `
[üì° PROXY LINK ESTABLISHED...]
`; log.scrollTop = log.scrollHeight; try { // This Proxy "hides" the request from your Wi-Fi filter const proxyUrl = "https://api.allorigins.win"; const wikiUrl = `https://en.wikipedia.org{encodeURIComponent(input)}`; const response = await fetch(proxyUrl + encodeURIComponent(wikiUrl)); const json = await response.json(); const data = JSON.parse(json.contents); // Proxies wrap the data in a 'contents' string const pages = data.query.pages; const pageId = Object.keys(pages)[0]; const extract = pages[pageId].extract; document.getElementById(loadingId).remove(); if (extract) { const shortText = extract.split('.').slice(0, 3).join('.') + '.'; log.innerHTML += `

`; } else { log.innerHTML += `
No records found for "${input}".
`; } } catch (error) { document.getElementById(loadingId).innerText = "‚ùå CRITICAL ERROR: All Satellite links blocked by Network."; } log.scrollTop = log.scrollHeight; }
